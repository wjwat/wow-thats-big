Wow, That's Big!

An API to return a single item for you to determine the size of it in bananas!

  • Weight is in bags of cement (50 kilograms)
  • Length is in Valve Hammers (16 Hammers = 1 foot)

// jQuery.ajax example
var getItem = $.get('https://wowthatsbig.net/api'),
    fillContainer = function(html) {
      $('#some-awesome-container').html(html);
    },
    oops = function() {
      console.log('Where did the item go?');
    };

getItem.then(fillContainer, oops);


// Vanilla JavaScript example
var xhr = new XMLHttpRequest();

xhr.open('GET', 'https://wowthatsbig.net/api');
xhr.onload = function() {
  if (this.status === 200) {
    document.querySelector('#some-awesome-container').innerHTML = this.response;
  } else {
    console.log('Where did the item go?');
  }
};
xhr.send();



// Using Fetch API
fetch('https://wowthatsbig.net/api')
  .then(response => response.json())
  .then(data => console.log(data))
  .catch(err => console.error('Where did the item go?'))
